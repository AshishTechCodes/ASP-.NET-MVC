5 Data Access Layer - Models/EmployeeDataAccessLayer.cs

using System.Data.SqlClient;

namespace EmployeeCRUD.Models;

public class EmployeeDataAccessLayer
{
    private static IConfiguration _configuration;

    public static void Configure(IConfiguration configuration)
    {
        _configuration = configuration;
    }

    private static string GetConnectionString()
    {
        return _configuration.GetConnectionString("EmployeeDBConnectionString");
    }

    // CREATE Operation
    public static void InsertEmployee(Employee employee)
    {
        string query = @"INSERT INTO Employee (FirstName, LastName, Email, Department, HireDate) 
                         VALUES (@FirstName, @LastName, @Email, @Department, @HireDate)";

        using (SqlConnection connection = new SqlConnection(GetConnectionString()))
        {
            using (SqlCommand command = new SqlCommand(query, connection))
            {
                command.Parameters.AddWithValue("@FirstName", employee.FirstName);
                command.Parameters.AddWithValue("@LastName", employee.LastName);
                command.Parameters.AddWithValue("@Email", employee.Email);
                command.Parameters.AddWithValue("@Department", employee.Department ?? (object)DBNull.Value);
                command.Parameters.AddWithValue("@HireDate", employee.HireDate ?? (object)DBNull.Value);
                
                connection.Open();
                command.ExecuteNonQuery();
            }
        }
    }

    // READ Operation - Get All Employees
    public static List<Employee> GetAllEmployees()
    {
        List<Employee> employeesList = new List<Employee>();
        string query = "SELECT EmployeeID, FirstName, LastName, Email, Department, HireDate FROM Employee";

        using (SqlConnection connection = new SqlConnection(GetConnectionString()))
        {
            using (SqlCommand command = new SqlCommand(query, connection))
            {
                connection.Open();
                using (SqlDataReader reader = command.ExecuteReader())
                {
                    while (reader.Read())
                    {
                        Employee employee = new Employee
                        {
                            EmployeeId = Convert.ToInt32(reader["EmployeeID"]),
                            FirstName = reader["FirstName"].ToString(),
                            LastName = reader["LastName"].ToString(),
                            Email = reader["Email"].ToString(),
                            Department = reader["Department"] == DBNull.Value ? null : reader["Department"].ToString(),
                            HireDate = reader["HireDate"] == DBNull.Value ? (DateTime?)null : Convert.ToDateTime(reader["HireDate"])
                        };
                        employeesList.Add(employee);
                    }
                }
            }
        }
        return employeesList;
    }
}
Key Point: SQL Parameters (@FirstName) prevent SQL injection attacks!

6 Create Controller - Controllers/EmployeeController.cs

using Microsoft.AspNetCore.Mvc;
using EmployeeCRUD.Models;

namespace EmployeeCRUD.Controllers;

public class EmployeeController : Controller
{
    // GET: Employee/Index - Display all employees
    public IActionResult Index()
    {
        var employees = EmployeeDataAccessLayer.GetAllEmployees();
        return View(employees);
    }

    // GET: Employee/Create - Show Create Form
    public IActionResult Create()
    {
        return View();
    }

    // POST: Employee/Create - Process Form Submission
    [HttpPost]
    [ValidateAntiForgeryToken]
    public IActionResult Create(Employee employee)
    {
        try
        {
            if (ModelState.IsValid)
            {
                EmployeeDataAccessLayer.InsertEmployee(employee);
                TempData["SuccessMessage"] = "Employee created successfully!";
                return RedirectToAction("Index");
            }
        }
        catch (Exception ex)
        {
            ModelState.AddModelError("", "Unable to save changes: " + ex.Message);
        }
        return View(employee);
    }
}
Attributes: [HttpPost] restricts to POST requests; [ValidateAntiForgeryToken] prevents CSRF attacks.

7 Configure Program.cs

using EmployeeCRUD.Models;

var builder = WebApplication.CreateBuilder(args);
builder.Services.AddControllersWithViews();

var app = builder.Build();

// Configure EmployeeDataAccessLayer
EmployeeDataAccessLayer.Configure(app.Configuration);

if (!app.Environment.IsDevelopment())
{
    app.UseExceptionHandler("/Home/Error");
    app.UseHsts();
}

app.UseHttpsRedirection();
app.UseStaticFiles();
app.UseRouting();
app.UseAuthorization();

// Changed default route to Employee controller
app.MapControllerRoute(
    name: "default",
    pattern: "{controller=Employee}/{action=Create}/{id?}");

app.Run();
Route Pattern: {controller=Employee}/{action=Create}/{id?} - Maps URL to EmployeeController by default.

Quick Reference: Predefined vs. Custom

Predefined (ASP.NET Core)
  • Controller - Base class
  • IActionResult - Action return type
  • [HttpPost] - Attribute
  • TempData - One-request data storage
  • ModelState - Validation container
  • @Html.BeginForm() - Form helper
Custom (You Create)
  • Employee - Model class
  • EmployeeController - Controller
  • EmployeeDataAccessLayer - Database logic
  • Create.cshtml - Form view
  • Index.cshtml - List view
  • _Layout.cshtml - Master page
